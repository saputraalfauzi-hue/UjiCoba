Contoh: JavaScript Todo App

class TodoApp {
    constructor() {
        this.todos = JSON.parse(localStorage.getItem('todos')) || [];
        this.init();
    }
    
    init() {
        this.render();
        document.getElementById('addTodo').addEventListener('click', () => this.addTodo());
        document.getElementById('clearCompleted').addEventListener('click', () => this.clearCompleted());
    }
    
    addTodo() {
        const input = document.getElementById('todoInput');
        const text = input.value.trim();
        
        if (text) {
            this.todos.push({
                id: Date.now(),
                text: text,
                completed: false,
                createdAt: new Date().toISOString()
            });
            
            input.value = '';
            this.saveAndRender();
        }
    }
    
    toggleTodo(id) {
        this.todos = this.todos.map(todo => 
            todo.id === id ? {...todo, completed: !todo.completed} : todo
        );
        this.saveAndRender();
    }
    
    saveAndRender() {
        localStorage.setItem('todos', JSON.stringify(this.todos));
        this.render();
    }
    
    render() {
        const container = document.getElementById('todoList');
        container.innerHTML = this.todos.map(todo => `
            

                
                ${todo.text}
                ${new Date(todo.createdAt).toLocaleDateString()}
            

        `).join('');
        
        document.getElementById('stats').textContent = 
            `Total: ${this.todos.length} | Completed: ${this.todos.filter(t => t.completed).length}`;
    }
}

const app = new TodoApp();

Contoh: Async Weather App

class WeatherApp {
    constructor(apiKey) {
        this.apiKey = apiKey;
        this.baseUrl = 'https://api.openweathermap.org/data/2.5';
        this.cache = new Map();
        this.cacheDuration = 10 * 60 * 1000;
    }
    
    async getWeather(city) {
        const cacheKey = `weather_${city}`;
        const cached = this.cache.get(cacheKey);
        
        if (cached && Date.now() - cached.timestamp < this.cacheDuration) {
            return cached.data;
        }
        
        try {
            const response = await fetch(
                `${this.baseUrl}/weather?q=${city}&appid=${this.apiKey}&units=metric`
            );
            
            if (!response.ok) {
                throw new Error(`City not found: ${city}`);
            }
            
            const data = await response.json();
            const weatherData = {
                city: data.name,
                country: data.sys.country,
                temp: Math.round(data.main.temp),
                feelsLike: Math.round(data.main.feels_like),
                humidity: data.main.humidity,
                description: data.weather[0].description,
                icon: data.weather[0].icon,
                windSpeed: data.wind.speed
            };
            
            this.cache.set(cacheKey, {
                data: weatherData,
                timestamp: Date.now()
            });
            
            return weatherData;
        } catch (error) {
            console.error('Error fetching weather:', error);
            throw error;
        }
    }
    
    async getForecast(city) {
        try {
            const response = await fetch(
                `${this.baseUrl}/forecast?q=${city}&appid=${this.apiKey}&units=metric`
            );
            
            if (!response.ok) throw new Error('Forecast not available');
            
            const data = await response.json();
            return data.list.slice(0, 5).map(item => ({
                time: new Date(item.dt * 1000).toLocaleTimeString(),
                temp: Math.round(item.main.temp),
                description: item.weather[0].description
            }));
        } catch (error) {
            console.error('Error fetching forecast:', error);
            return [];
        }
    }
    
    displayWeather(data) {
        const html = `
            

                
${data.city}, ${data.country}

                

                    
                    

                        
${data.temp}°C

                        
${data.description}

                        
Feels like: ${data.feelsLike}°C

                        
Humidity: ${data.humidity}%

                        
Wind: ${data.windSpeed} m/s

                    

                

            

        `;
        
        document.getElementById('weatherContainer').innerHTML = html;
    }
}
