HTML Dasar (Minggu 1-2)

HTML adalah tulang punggung setiap website. Pelajari struktur dasar dan elemen penting.

Struktur Dokumen

DOCTYPE, html, head, body, meta tags

Heading & Teks

h1-h6, p, span, strong, em, br, hr

List & Tabel

ul, ol, li, table, tr, td, th

Link & Gambar

a, img, src, href, alt, target

Form & Input

form, input, textarea, select, button

Semantic HTML5

header, nav, main, section, article, footer

Contoh: Halaman HTML Lengkap

<!DOCTYPE html>
<html lang="id">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Belajar HTML</title>
</head>
<body>
    <header>
        <h1>Selamat Datang</h1>
        <nav>
            <a href="#home">Home</a>
            <a href="#about">Tentang</a>
        </nav>
    </header>
    <main>
        <article>
            <h2>Judul Artikel</h2>
            <p>Ini adalah paragraf pertama.</p>
            <img src="gambar.jpg" alt="Deskripsi gambar">
        </article>
    </main>
    <footer>
        <p>Hak Cipta © 2023</p>
    </footer>
</body>
</html>

    display: flex;
    flex-wrap: wrap;

    align-items: center;

}

.flex-item {
    flex: 1 1 300px;
    min-height: 200px;
    background: #eee;
    border-radius: 8px;
    padding: 20px;
}

/* CSS Grid Layout */
.grid-container {
    display: grid;
    grid-template-columns: repeat(auto-fit, minmax(250px, 1fr));
    gap: 30px;
    padding: 20px;
}

.grid-item {
    background: white;
    border-radius: 10px;
    overflow: hidden;
    box-shadow: 0 4px 8px rgba(0,0,0,0.1);
}

/* Animasi */

    from { opacity: 0; transform: translateY(20px); }
    to { opacity: 1; transform: translateY(0); }
}

.animated-item {
    animation: fadeIn 0.5s ease-out;

const [first, second, ...rest] = [1, 2, 3, 4, 5];

// Async/Await
async function ambilData() {
    try {
        const response = await fetch('https://api.example.com/data');
        if (!response.ok) {
            throw new Error('Network response was not ok');
        }
        const data = await response.json();
        console.log('Data diterima:', data);
        return data;
    } catch (error) {
        console.error('Error mengambil data:', error);
        throw error;
    }
}

// Classes
class Person {
    constructor(nama, umur) {
        this.nama = nama;
        this.umur = umur;
    }
    
    sapa() {
        return `Halo, nama saya ${this.nama}`;
    }
    
    static info() {
        return 'Ini adalah class Person';
    }
}

class Developer extends Person {
    constructor(nama, umur, bahasa) {
        super(nama, umur);
        this.bahasa = bahasa;
    }
    
    coding() {
        return `${this.nama} sedang coding dengan ${this.bahasa}`;
    }
}

const dev = new Developer('Ali', 25, 'JavaScript');

Contoh: React Component

import React, { useState, useEffect } from 'react';

function Counter() {
    const [count, setCount] = useState(0);
    const [message, setMessage] = useState('');
    
    useEffect(() => {
        if (count > 10) {
            setMessage('Wow, angka sudah besar!');
        } else {
            setMessage('');
        }
    }, [count]);
    
    const increment = () => setCount(count + 1);
    const decrement = () => setCount(count - 1);
    const reset = () => setCount(0);
    
    return (
        <div className="counter">
            <h2>Counter: {count}</h2>
            <p>{message}</p>
            <div className="buttons">
                <button onClick={increment}>Tambah (+)</button>
                <button onClick={decrement}>Kurang (-)</button>
                <button onClick={reset}>Reset</button>
            </div>
            <DisplayCount count={count} />
        </div>
    );
}

function DisplayCount({ count }) {
    return (
        <div className="display">
            <p>Angka saat ini: <strong>{count}</strong></p>
            <p>{count % 2 === 0 ? 'Genap' : 'Ganjil'}</p>
        </div>
    );
}

export default Counter;
